FDCPE_Estado_FFd1: FDCPE port map (Estado_FFd1,Estado_FFd1_D,clk,'0','0');
     Estado_FFd1_D <= ((en AND Estado_FFd3 AND Estado_FFd2)
      OR (NOT en AND NOT Estado_FFd3 AND Estado_FFd2));
FDCPE_Estado_FFd2: FDCPE port map (Estado_FFd2,Estado_FFd2_D,clk,'0','0');
     Estado_FFd2_D <= ((Estado_FFd1)
      OR (en AND Estado_FFd3 AND NOT Estado_FFd2)
      OR (NOT en AND NOT Estado_FFd3 AND NOT Estado_FFd2));
FDCPE_Estado_FFd3: FDCPE port map (Estado_FFd3,Estado_FFd3_D,clk,'0','0');
     Estado_FFd3_D <= ((NOT en AND Estado_FFd1)
      OR (en AND Estado_FFd3 AND NOT Estado_FFd2)
      OR (en AND NOT Estado_FFd2 AND NOT Estado_FFd1)
      OR (NOT en AND Estado_FFd3 AND Estado_FFd2));
FDCPE_s0: FDCPE port map (s(0),s_D(0),clk,'0','0');
     s_D(0) <= ((Estado_FFd1)
      OR (Estado_FFd3 AND NOT Estado_FFd2));
FDCPE_s1: FDCPE port map (s(1),s_D(1),clk,'0','0');
     s_D(1) <= ((Estado_FFd1)
      OR (Estado_FFd3 AND Estado_FFd2));
FDCPE_s2: FDCPE port map (s(2),s_D(2),clk,'0','0');
     s_D(2) <= (NOT Estado_FFd3 AND Estado_FFd2);
